--------------------------------------------------------------------------------
\\Companies that propose help and also provide certifications.\\
--------------------------------------------------------------------------------
\begin{lstlisting}[language = SQL]
SELECT DISTINCT comb.CID, COMPANY.Name
FROM(
	SELECT DISTINCT CID
	FROM SUPPORTING 

	UNION ALL 

	SELECT DISTINCT CID
	FROM CERTIFYING
) comb, COMPANY
WHERE comb.CID = COMPANY.CID
\end{lstlisting}
