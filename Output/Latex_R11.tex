--------------------------------------------------------------------------------
\\Types of actions without funding program.\\
--------------------------------------------------------------------------------
\begin{lstlisting}[language = SQL]
SELECT ACTIONTYPE.TID, ACTIONTYPE.DESCRIPTION
FROM ACTIONTYPE 
WHERE ACTIONTYPE.TID NOT IN(
SELECT ACTION.TID FROM
ACTION, ACTIONFUND
WHERE ACTION.AID = ACTIONFUND.AID)
\end{lstlisting}
